// RA 2.1 — Herencia + Polimorfismo + Sobrecarga
class Cuenta {
  protected: string titular; double saldo;
  public:
    virtual void aplicarBeneficio() = 0;  // virtual puro
    Cuenta operator+(Cuenta& otra) { return Cuenta(saldo + otra.saldo); }
};

class CuentaAhorro : public Cuenta {  // Herencia simple
  void aplicarBeneficio() override { saldo *= 1.10; }  // Polimorfismo
};

// RA 2.2 — Patrón Singleton + Excepciones
class GestorCajero {
  static GestorCajero* instancia;  // Singleton: una sola instancia
  GestorCajero() {}  // constructor privado
  public:
    static GestorCajero* getInstance() {
      if (!instancia) instancia = new GestorCajero();
      return instancia;
    }
};
